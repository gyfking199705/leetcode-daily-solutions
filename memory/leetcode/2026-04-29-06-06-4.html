LeetCode #4

寻找两个正序数组的中位数 · Median of Two Sorted Arrays

📝 题目描述

给定两个大小分别为 m 和 n 的正序（从小到大）数组 nums1 和 nums2。请你找出并返回这两个正序数组的 中位数。


算法的时间复杂度应该为 O(log (m+n))。

示例 1
输入：nums1 = [1,3], nums2 = [2] 输出：2.00000 解释：合并数组 = [1,2,3] ，中位数 2
示例 2
输入：nums1 = [1,2], nums2 = [3,4] 输出：2.50000 解释：合并数组 = [1,2,3,4] ，中位数 (2 + 3) / 2 = 2.5

💡 核心思路

  • 1
    问题转化：找第k小元素

    中位数 = 合并后数组的中间元素。奇数长度找第 (m+n)/2+1 个，偶数长度找第 (m+n)/2 和 (m+n)/2+1 的平均值。

  • 2
    二分排除法

    要找到第k小的元素，比较 nums1[k/2-1] 和 nums2[k/2-1]，较小的那一半不可能包含第k小元素，可以排除。

  • 3
    递归缩小问题规模

    每次排除 k/2 个元素，k 减半，时间复杂度 O(log(m+n))。

class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        total = len(nums1) + len(nums2)
        
        # 奇数：找第 (total//2 + 1) 个
        # 偶数：找第 total//2 和 total//2 + 1 个的平均值
        if total % 2 == 1:
            return self.getKth(nums1, nums2, total // 2 + 1)
        else:
            left = self.getKth(nums1, nums2, total // 2)
            right = self.getKth(nums1, nums2, total // 2 + 1)
            return (left + right) / 2
    
    def getKth(self, nums1, nums2, k):
        # 确保 nums1 是较短的数组，减少边界情况
        if len(nums1) > len(nums2):
            return self.getKth(nums2, nums1, k)
        
        # 边界情况
        if len(nums1) == 0:
            return nums2[k - 1]
        if k == 1:
            return min(nums1[0], nums2[0])
        
        # 比较第 k//2 个元素，排除较小的一半
        i = min(k // 2, len(nums1))
        j = min(k // 2, len(nums2))
        
        if nums1[i - 1] < nums2[j - 1]:
            # nums1 的前 i 个元素不可能包含第 k 小
            return self.getKth(nums1[i:], nums2, k - i)
        else:
            # nums2 的前 j 个元素不可能包含第 k 小
            return self.getKth(nums1, nums2[j:], k - j)

class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int m = nums1.length, n = nums2.length;
        int total = m + n;
        
        if (total % 2 == 1) {
            return getKth(nums1, 0, nums2, 0, total / 2 + 1);
        } else {
            int left = getKth(nums1, 0, nums2, 0, total / 2);
            int right = getKth(nums1, 0, nums2, 0, total / 2 + 1);
            return ((double) left + right) / 2;
        }
    }
    
    // 找第 k 小的元素（k 从 1 开始）
    private int getKth(int[] nums1, int i, int[] nums2, int j, int k) {
        if (i >= nums1.length) return nums2[j + k - 1];
        if (j >= nums2.length) return nums1[i + k - 1];
        if (k == 1) return Math.min(nums1[i], nums2[j]);
        
        int mid1 = Math.min(k / 2, nums1.length - i);
        int mid2 = Math.min(k / 2, nums2.length - j);
        
        if (nums1[i + mid1 - 1] < nums2[j + mid2 - 1]) {
            return getKth(nums1, i + mid1, nums2, j, k - mid1);
        } else {
            return getKth(nums1, i, nums2, j + mid2, k - mid2);
        }
    }
}

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size(), n = nums2.size();
        int total = m + n;
        
        if (total % 2 == 1) {
            return getKth(nums1, 0, nums2, 0, total / 2 + 1);
        } else {
            int left = getKth(nums1, 0, nums2, 0, total / 2);
            int right = getKth(nums1, 0, nums2, 0, total / 2 + 1);
            return (left + right) / 2.0;
        }
    }
    
private:
    int getKth(vector<int>& nums1, int i, 
               vector<int>& nums2, int j, int k) {
        if (i >= nums1.size()) return nums2[j + k - 1];
        if (j >= nums2.size()) return nums1[i + k - 1];
        if (k == 1) return min(nums1[i], nums2[j]);
        
        int mid1 = min(k / 2, (int)nums1.size() - i);
        int mid2 = min(k / 2, (int)nums2.size() - j);
        
        if (nums1[i + mid1 - 1] < nums2[j + mid2 - 1]) {
            return getKth(nums1, i + mid1, nums2, j, k - mid1);
        } else {
            return getKth(nums1, i, nums2, j + mid2, k - mid2);
        }
    }
};

🎯 面试连环追问

Q1：为什么时间复杂度要求是 O(log(m+n))？ ▼
这提示我们需要使用二分查找。如果直接合并数组是 O(m+n)，如果用二分查找找到第k小元素则是 O(log(m+n))。这是对算法选择的重要提示。
Q2：如何证明每次排除 k/2 个元素是正确的？ ▼
假设 nums1[k/2-1] <= nums2[k/2-1]，那么 nums1 的前 k/2 个元素最多是第 k/2 小，不可能成为第 k 小。同时 nums1[k/2-1] 及之前的元素最多有 k/2-1 个比它小，即使加上 nums2 中所有比它小的，也最多 k-1 个，所以第 k 小一定在后面。
Q3：能否用迭代代替递归？ ▼
可以。递归深度是 O(log(m+n))，一般不会栈溢出，但也可以用迭代实现。思路是用循环代替递归，维护两个指针和当前要找的 k 值，直到 k=1 或一个数组被排除完。
Q4：如果要求 O(log(min(m,n))) 的解法？ ▼
另一种解法是在较短的数组上做二分。将 nums1 分成两部分，根据 nums1 的分割点确定 nums2 的分割点，使得左边元素总数等于右边（或差1），然后检查边界条件是否满足。如果不满足就调整分割点。这是更优的解法，面试中作为加分项。

📊 复杂度分析

时间复杂度

每次递归 k 减半，最多递归 O(log(m+n)) 次，所以时间复杂度为 O(log(m+n))

空间复杂度

递归栈深度为 O(log(m+n))。若用迭代可优化至 O(1)
